مخرج حروف المد الثلاثة (الأصلي)

خروج الألف والواو والياء المدية من تجويف الحلق والفم دون تصادم أو نقطة ارتكاز محددة.
ببساطة:هنطق الحرف ممتد مع الهواء، ولساني مرتاح تماماً لتحت!
امتى؟ألف ساكنة قبلها فتح [ كَانَ ]، واو ساكنة قبلها ضم [ يَقُولُ ]، ياء ساكنة قبلها كسر [ قِيلَ ].

خلاء الفم والحلق
الآيات التطبيقية الخمس (بصوت الشيخ الحصري):
الألف
مَالِكِ يَوْمِ الدِّينِ ﴿٤﴾
الواو

الياء
فِي جِيدِهَا حَبْلٌ مِّن مَّسَدٍ ﴿٥﴾
الألف
تَبَّتْ يَدَا أَبِي لَهَبٍ وَتَبَّ ﴿١﴾
الواو
قُلْ يَا أَيُّهَا الْكَافِرُونَ ﴿١﴾

أقصى الحلق (الهمزة والهاء)

خروج الحرف من أبعد منطقة في الحلق مما يلي الصدر (منطقة الأوتار الصوتية).
ببساطة:الحرف بيطلع من أعمق حتة في الزور تحت خالص قريبة من صدرك!
امتى؟عند النطق بالهمزة [ أ ] مخرجا صريحاً، أو الهاء [ هـ ] باهتزاز الأوتار.
الآيات التطبيقية الخمس (بصوت الشيخ الحصري):
الهمزة
قُلْ هُوَ اللَّهُ أَحَدٌ ﴿١﴾
الهاء
قُلْ هُوَ اللَّهُ أَحَدٌ ﴿١﴾
الهاء
الَّذِي أَطْعَمَهُم مِّن جُوعٍ وَآمَنَهُم مِّنْ خَوْفٍ ﴿٤﴾
الهمزة
.. وَأَرْسَلَ عَلَيْهِمْ طَيْرًا أَبَابِيلَ ﴿٣﴾
الهمزة
وَلَمْ يَكُن لَّهُ كُفُوًا أَحَدٌ ﴿٤﴾

وسط الحلق (العين والحاء)

خروج الحرف من منطقة لسان المزمار (منتصف الرقبة تماماً).
ببساطة:الحرف بيطلع من نص زورك بالظبط بضغط لسان المزمار لورا!
العين
الْحَمْدُ لِلَّهِ رَبِّ الْعَالَمِينَ ﴿٢﴾
الحاء
الْحَمْدُ لِلَّهِ رَبِّ الْعَالَمِينَ ﴿٢﴾
العين
خَلَقَ الْإِنسَانَ مِنْ عَلَقٍ ﴿٢﴾
الحاء
فَسَبِّحْ بِحَمْدِ رَبِّكَ وَاسْتَغْفِرْهُ ﴿٣﴾
العين
إِيَّاكَ نَعْبُدُ وَإِيَّاكَ نَسْتَعِينُ ﴿٥﴾

أدنى الحلق (الغين والخاء)

خروج الحرف من أقرب منطقة في الحلق إلى الفم (منطقة جذر اللسان مع الحنك اللحمي).
ببساطة:أعلى حتة في زورك قريبة جداً من بقك من ورا!
الخاء
وَآمَنَهُم مِّن خَوْفٍ ﴿٤﴾
الغين
غَيْرِ الْمَغْضُوبِ عَلَيْهِمْ ﴿٧﴾
الغين
الْمَغْضُوبِ عَلَيْهِمْ ﴿٧﴾
الخاء
خَلَقَ الْإِنسَانَ مِنْ عَلَقٍ ﴿٢﴾
الخاء
أَلَمْ يَعْلَم بِأَنَّ اللَّهَ يَرَىٰ ﴿١٤﴾

مخرج حرف القاف [ ق ]

أقصى اللسان (أبعده من وراء) مع ما يقابله من الحنك اللحمي (الرخو).
ببساطة:آخر لسانك من ورا بيقفل تماماً مع الحنك اللحمي الطري فوق، والحرف بيطلع تخين ومقلقل!
القاف
الْقَارِعَةُ مَا الْقَارِعَةُ ﴿١﴾
القاف
قُلْ أَعُوذُ بِرَبِّ الْفَلَقِ ﴿١﴾
القاف
لَقَدْ خَلَقْنَا الْإِنسَانَ فِي أَحْسَنِ تَقْوِيمٍ ﴿٤﴾
القاف
بِأَيِّ ذَنبٍ قُتِلَتْ ﴿٩﴾
القاف
إِذَا السَّمَاءُ انفَطَرَتْ ﴿١﴾

مخرج حرف الكاف [ ك ]

أقصى اللسان بعد مخرج القاف قليلاً (إلى الأمام فميّاً) مع الحنك العظمي واللحمي معاً.
ببساطة:آخر لسانك بيقفل بردو بس قدام مخرج القاف بحتة صغيرة عند الجزء العظمي الناشف!
الكاف
إِنَّا أَعْطَيْنَاكَ الْكَوْثَرَ ﴿١﴾
الكاف
أَلَمْ نَخْلُقكُّم مِّن مَّاءٍ مَّهِينٍ ﴿٢٠﴾
الكاف
أَلْهَاكُمُ التَّكَاثُرُ ﴿١﴾
الكاف
إِذَا السَّمَاءُ كَانَتْ وَرْدَةً كَالدِّهَانِ ﴿٣٧﴾
الكاف
وَلَمْ يَكُن لَّهُ كُفُوًا أَحَدٌ ﴿٤﴾

وسط اللسان (الجيم والشين والياء)

وسط اللسان مع ما يقابله من وسط الحنك الأعلى (غار الحنك).
ببساطة:نص لسانك بيترفع لفوق ويصطدم بسقف الحنك العظمي (مجموعة في كلمة جيش)!
الجيم
إِذَا جَاءَ نَصْرُ اللَّهِ وَالْفَتْحُ ﴿١﴾
الشين
مِن شَرِّ مَا خَلَقَ ﴿٢﴾
الياء
مَالِكِ يَوْمِ الدِّينِ ﴿٤﴾
الجيم
فِي جِيدِهَا حَبْلٌ مِّن مَّسَدٍ ﴿٥﴾
الشين
إِيلَافِهِمْ رِحْلَةَ الشِّتَاءِ وَالصَّيْفِ ﴿٢﴾

طرف اللسان (الطاء والدال والتاء)

طرف اللسان مع أصول الثنايا العليا (منبت الأسنان الأمامية فوق).
ببساطة:بوز لسانك من قدام بيخبط في اللثة العريضة منبت الأسنان اللي فوق!
الطاء
وَأَرْسَلَ عَلَيْهِمْ طَيْرًا أَبَابِيلَ ﴿٣﴾
الدال
قُلْ هُوَ اللَّهُ أَحَدٌ ﴿١﴾
التاء
تَبَّتْ يَدَا أَبِي لَهَبٍ وَتَبَّ ﴿١﴾
الطاء
وَلَا يَحُضُّ عَلَىٰ طَعَامِ الْمِسْكِينِ ﴿٣﴾
الدال
لَمْ يَلِدْ وَلَمْ يُولَدْ ﴿٣﴾

حروف الصفير (الصاد والسين والزاي)

طرف اللسان مع ما فوق الثنايا السفلى (بين الأسنان العليا والسفلى).
ببساطة:الصاد: ضهر لسانك بيترفع لسقف البق والبق بيتمست بصدى تخين! السين: لسانك مرتاح لتحت والصوت رفيع!
الصاد
صِرَاطَ الَّذِينَ أَنْعَمْتَ عَلَيْهِمْ ﴿٧﴾
السين
الَّذِينَ هُمْ عَن صَلَاتِهِمْ سَاهُونَ ﴿٥﴾
السين
فَسَبِّحْ بِحَمْدِ رَبِّكَ وَاسْتَغْفِرْهُ ﴿٣﴾
الصاد
وَالَّذِينَ آمَنُوا وَعَمِلُوا الصَّالِحَاتِ ﴿٢٥﴾
السين
بِسْمِ اللَّهِ الرَّحْمَٰنِ الرَّحِيمِ ﴿١！】

مخرج الشفتين بالكامل

انطباق الشفتين (الباء والميم)، انضمامهما (الواو)، بطن الشفة مع الثنايا العليا (الفاء).
ببساطة:الشفايف بتقفل خالص في الباء والميم، وبتتضم في الواو، وسنانك اللي فوق بتخبط تحت في الفاء!
الباء
تَبَّتْ يَدَا أَبِي لَهَبٍ وَتَبَّ ﴿١﴾
الميم
مَالِكِ يَوْمِ الدِّينِ ﴿٤﴾
الفاء
أَلَمْ تَرَ كَيْفَ فَعَلَ رَبُّكَ بِأَصْحَابِ الْفِيلِ ﴿١﴾
الواو
قُلْ هُوَ اللَّهُ أَحَدٌ ﴿١﴾
الميم
الَّذِي أَطْعَمَهُم مِّـن جُوعٍ ﴿٤﴾

مخرج الخيشوم (صوت الغنة حركتين)

الخيشوم هو التجويف الأنفى الداخلي الممتد فوق غار الحنك، وهو مخرج صفة الغنة لا حروف مستقلة.
ببساطة:صوت رخيم زمني بيطلع من مناخيرك ميكملش نطق النون والميم المشددتين من غيره!
غنة نون
فَإِنَّ مَعَ الْعُسْرِ يُسْرًا ﴿٥﴾
غنة ميم
عَمَّ يَتَسَاءَلُونَ ﴿١﴾
غنة نون
مِنَ الْجِنَّةِ وَالنَّاسِ ﴿٦﴾
غنة ميم
فَأَمَّا مَنْ أَعْطَىٰ وَاتَّقَىٰ ﴿٥﴾
غنة نون
إِنَّا أَعْطَيْنَاكَ الْكَوْثَرَ ﴿١﴾
